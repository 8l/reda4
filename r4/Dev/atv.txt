|----------------------------------------
:sprint | "" --
	mprint
	ccx ccy ink@
	2 2 +atxy pick3
	tshadow ink printx
	ink atxy
	printx ;

|----------------------------------------
:renderimg
	clrscr
	0 0 sw sh ifondo img.drawsize
|	0 0  ifondo img.draw
	droidsansbold 64 rfont!
	verde
	"Hola mundo !" print cr
	"¿ Como cuidarse con las esdrújulas?" print cr
	rojo
	"¿que esta pasando...?" sprint
    >XFB
	cls
	;

#xm 0
:marquesina
	xm sh 90 - atxy
	"esto es un texto mas largo que la pantalla para que pueda leerse a medida que va pasando etexto que es" sprint
	-2 'xm +!
	;
|----------------------------------------
:init
	mark
	"media/bmp/cerezop.png" loadpng 'ifondo !
	;

:main
	renderimg
	33
	sw 'xm !
	show xfb>scr
		verde
		dup "%d" sprint
		rojo
		sw 2/ 200 + sh 2/
		msec 3 << 300 xy+polar
		atxy
		msec "%d" sprint

		marquesina
		'exit >esc<
		;
