|	2 =? ( drop 'acoin msec 5 >> $1c and + ; )

:drawtile | x y -- x y
	over xp escalap >> +
	over yp escalap >> +
	[mapa] @ 0? ( drop ; )
	pick2 escalap << sw 2/ + xp $1f and -
	pick2 escalap << sh 2/ + yp $1f and -
	rot dibujotile
